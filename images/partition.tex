\matrix (A) [collection] {
    \node (a1) {$a_1$}; &
    \node (a2) {$a_2$}; &
    \node (a3) {$a_3$}; &
    \node [elements between]; &
    \node (ai-1) {$a_{i-1}$}; &
    \node (ai) {$a_i$}; &
    \node [elements between]; &
    \node (an-1) {$a_{n-1}$}; &
    \node (an) {$a_n$}; \\
};

\matrix [tuple of collections, below=2 of A] {
    \node (b1) {$a_1$}; &
    \node (b3) {$a_3$}; &
    \node [elements between]; &
    \node (bi-1) {$a_{i-1}$}; &
    \node [elements between]; &
    \node (bn-1) {$a_{n-1}$}; &
    \node [tuple separator]; &
    \node (b2) {$a_2$}; &
    \node [elements between]; &
    \node (bi) {$a_i$}; &
    \node [elements between]; &
    \node (bn) {$a_n$}; \\
};

\foreach \i/\r in {1/true,2/false,3/true,i-1/true,i/false,n-1/true,n/false}{
    \node (f) [subfunction, above=.5 of a\i] {\texttt{p}};
    \node (r) [subelement, above=.5 of f] {\texttt{\r}};
    \draw [subflow ->] (a\i) -- (f.south io 1);
    \draw [subflow ->] (f.north io 1) -- (r);
}
